{# Componente: Breadcrumbs Descrição: Navegação breadcrumb reutilizável Parâmetros: - items: Lista de items do breadcrumb [{'label': 'Texto', 'url': '/path'}] O último item não deve ter URL (será o item ativo) Uso: {% from 'components/breadcrumbs.html' import breadcrumbs %} {{ breadcrumbs([ {'label': 'Home', 'url': '/'}, {'label': 'Produtos', 'url': '/produtos'}, {'label': 'Editar Produto'} ]) }} #} {% macro breadcrumbs(items) %}
    {% for item in items %}
  1. {% if not loop.last and item.url %} {{ item.label }} {% else %} {{ item.label }} {% endif %}
    2. {% endfor %}
{% endmacro %}
